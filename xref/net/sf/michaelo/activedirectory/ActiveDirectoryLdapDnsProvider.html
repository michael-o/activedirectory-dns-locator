16  package net.sf.michaelo.activedirectory;
17  
18  import java.net.URI;
19  import java.net.URISyntaxException;
20  import java.util.ArrayList;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.Map.Entry;
24  import java.util.Optional;
25  
26  import javax.naming.NamingException;
27  import com.sun.jndi.ldap.spi.LdapDnsProvider;
28  import com.sun.jndi.ldap.spi.LdapDnsProviderResult;
29  
30  import net.sf.michaelo.activedirectory.ActiveDirectoryDnsLocator.HostPort;
31  
32  import org.apache.commons.lang3.Validate;
33  
34  /**
35   * A {@link LdapDnsProvider} for the Active Directory. This implementation hooks into Sun's/Oracle's
36   * LDAP implementation with JNDI to autodiscover Active Directory servers for LDAP via DNS.
37   * <br>
38   * This provider receives all environment properties via JNDI. It recornizes the following properties:
39   * <ul>
40   * <li>{@code net.sf.michaelo.activedirectory.site}: The Active Directory site your client resides.</li>
41   * <li>{@code dns.*}: Passthrough properties to the underlying JNDI DNS Provider. (prefix will be stripped)</li>
42   * </ul>
43   */
44  @SuppressWarnings("restriction")
45  public class ActiveDirectoryLdapDnsProvider extends LdapDnsProvider {
46  
47  	public static final String SITE_PROPERTY = "net.sf.michaelo.activedirectory.site";
48  
49  	@Override
50  	public Optional<LdapDnsProviderResult> lookupEndpoints(String url, Map<?, ?> env)
51  			throws NamingException {
52  		Validate.notEmpty(url, "url cannot be null or empty");
53  		Validate.notNull(env, "env cannot be null");
54  
55  		URI ldapUri = null;
56  
57  		try {
58  			ldapUri = URI.create(url);
59  		} catch (IllegalArgumentException e) {
60  			NamingException ne = new NamingException("URL '" + url + "' is invalid");
61  			ne.initCause(e);
62  			throw ne;
63  		}
64  
65  		ActiveDirectoryDnsLocator.Builder builder = new ActiveDirectoryDnsLocator.Builder();
66  		Map<String, Object> typedEnv = (Map<String, Object>) env;
67  		for (Entry<String, Object> entry : typedEnv.entrySet()) {
68  			String key = entry.getKey();
69  			if (key.startsWith("dns.")) {
70  				key = key.substring(4);
71  				builder.additionalProperty(key, entry.getValue());
72  			}
73  		}
74  
75  		ActiveDirectoryDnsLocator locator = builder.build();
76  
77  		String domainName = ldapUri.getHost();
78  		String site = (String) env.get(SITE_PROPERTY);
79  
80  		HostPort[] hostPorts = locator.locate("ldap", site, domainName);
81  
82  		if (hostPorts != null) {
83  			List<String> endpoints = new ArrayList<>(hostPorts.length);
84  
85  			for (HostPort hostPort : hostPorts) {
86  				try {
87  					URI endpointUri = new URI(ldapUri.getScheme(), null, hostPort.getHost(),
88  							ldapUri.getPort(), ldapUri.getPath(), ldapUri.getQuery(),
89  							ldapUri.getFragment());
90  					endpoints.add(endpointUri.toASCIIString());
91  				} catch (URISyntaxException e) {
92  					NamingException ne = new NamingException("failed to construct endpoint URL");
93  					ne.initCause(e);
94  					throw ne;
95  				}
96  			}
97  
98  			return Optional.of(new LdapDnsProviderResult(domainName, endpoints));
99  		}
100 
101 		return Optional.empty();
102 	}
103 
104 }
